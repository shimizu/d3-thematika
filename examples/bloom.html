Bloomエフェクトの使用例

// effect-utils.tsを使用したBloomエフェクトの実装
// UMDビルドではThematikaオブジェクトから直接アクセス

// カスタムBloomフィルターの作成
const bloomFilter = Thematika.createBloom({
    id: 'custom-bloom',
    intensity: 5,      // ブラーの強度
    threshold: 0.5,    // 明度閾値
    color: '#ffffff'   // 発光色
});

// SVGのdefsにフィルターを追加
const svg = d3.select('#map svg');
const defs = svg.append('defs');
bloomFilter(defs);

// GeojsonLayerにフィルターを適用
const worldLayer = new Thematika.GeojsonLayer({
    data: geojson,
    attr: {
        'fill': (d, i) => {
            // 明るい色を使用してBloom効果を強調
            const colors = ['#00ffff', '#ff00ff', '#ffff00', '#00ff00', '#ff6b6b'];
            return colors[i % colors.length];
        },
        'filter': bloomFilter.url(),  // Bloomフィルターを適用
        'stroke': '#ffffff',
        'stroke-width': 1,
        'opacity': 0.9
    }
});

// プリセットフィルターの使用例
const standardBloom = Thematika.FilterPresets.standardBloom();
const strongBloom = Thematika.FilterPresets.strongBloom();
