Basic Setup
Adding Layers
Styling
// 基本的なセットアップ
const map = new Thematika.Map({
    container: '#map',
    width: 800,
    height: 600,
    projection: d3.geoNaturalEarth1()
});

// GeoJSONデータの読み込み
const geojson = await d3.json("world.geojson");
// レイヤーの追加
const worldLayer = new Thematika.GeojsonLayer({
    data: geojson,
    attr: {
        fill: '#3498db',
        stroke: '#2c3e50',
        'stroke-width': 0.5
    }
});

const graticuleLayer = new Thematika.GraticuleLayer({
    step: [10, 10],
    attr: {
        stroke: '#ddd',
        'stroke-width': 0.5
    }
});

map.addLayer('world', worldLayer);
map.addLayer('graticule', graticuleLayer);
// 動的なスタイリング
const colorScale = d3.scaleSequential(d3.interpolateViridis)
    .domain([0, 100]);

const styledLayer = new Thematika.GeojsonLayer({
    data: geojson,
    attr: {
        fill: (d) => colorScale(d.properties.value),
        opacity: 0.8,
        cursor: 'pointer'
    },
    on: {
        mouseover: function(event, d) {
            d3.select(this)
                .transition()
                .duration(200)
                .attr('opacity', 1);
        }
    }
});
