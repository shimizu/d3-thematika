LineEdgeBundlingLayerの使用例

// LineEdgeBundlingLayerの使用例
const lineData = {
    "type": "FeatureCollection",
    "features": [
        {
            "type": "Feature",
            "geometry": {
                "type": "LineString",
                "coordinates": [[-120, 40], [-75, 35]]
            },
            "properties": { "name": "route1" }
        },
        {
            "type": "Feature",
            "geometry": {
                "type": "LineString",
                "coordinates": [[-118, 38], [-77, 37]]
            },
            "properties": { "name": "route2" }
        }
    ]
};

const width = 800;
const height = 600;

const projection = d3.geoAlbersUsa()
    .scale(1000)
    .translate([width / 2, height / 2]);

const map = new Thematika.Map({
    container: '#map',
    width: width,
    height: height,
    projection: projection
});

// LineEdgeBundlingLayerインスタンスを作成
const bundlingLayer = new Thematika.LineEdgeBundlingLayer({
    data: lineData,
    bundlingStrength: 0.85,
    forceStrength: 20,
    showOriginalLines: false,
    showControlPoints: false,
    animateForce: true,
    attr: {
        "stroke": '#ff6b6b',
        "stroke-width": 2,
        "opacity": 0.7
    }
});

// イベントハンドラーを追加
bundlingLayer.on('click', (event, data) => {
    console.log('クリックされたライン:', data.feature.properties.name);
});

// レイヤーを地図に追加
map.addLayer('bundling', bundlingLayer);
