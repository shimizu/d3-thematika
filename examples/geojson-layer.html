GeojsonLayerの使用例

// GeojsonLayerの使用例
const geojson = await d3.json("./geojson/world.geojson");

const width = 800;
const height = 600;

const projection = d3.geoEquirectangular()
    .fitExtent([[10, 0], [width-10, height]], geojson);

const map = new Thematika.Map({
    container: '#map',
    width: width,
    height: height,
    projection: projection
});

// GeojsonLayerインスタンスを作成
const worldLayer = new Thematika.GeojsonLayer({
    data: geojson,                
    attr: { 
        "fill": (d, i) => {
            // Viridis パレット - 2色の美しい対比
            const colors = ['#440154', '#1f9d8a']; // ディープパープルとティールグリーン
            return colors[i % colors.length];
        },
        "filter": shadowFilter.url(), 
        "stroke": '#1e293b',
        "stroke-width": 0.3,
        "opacity": 0.85,
    }
});

// OutlineLayerインスタンスを作成（クリップパス機能ON）
const outlineLayer = new Thematika.OutlineLayer({
    createClipPath: true,
    clipPathId: 'earth-outline-clip',
    attr: {
        "fill": 'none',
        "stroke": '#475569',
        "stroke-width": 2,
        "opacity": 0.7
    }
});

// GraticuleLayerインスタンスを作成
const graticuleLayer = new Thematika.GraticuleLayer({
    step: [10, 10], // 10度間隔
    attr: {
        "fill": 'none',
        "stroke": 'rgba(148, 163, 184, 0.4)',
        "stroke-width": 0.3,
        "opacity": 0.6
    }
});

// レイヤーインスタンスを地図に追加
map.addLayer('graticule', graticuleLayer);
map.addLayer('outline', outlineLayer);
map.addLayer('world', worldLayer);
